# Move final data to app folder


```r
packages <- 
  c(
    "here"
  )

pacman::p_load(packages,
               character.only = TRUE)
```


```r
file.copy(
  list.files(
    here(
      "..",
      "data",
      "final"
    ),
    full.names = TRUE
  ),
  here(
    "..",
    "app",
    "data"
  ),
  recursive = TRUE
)
```
